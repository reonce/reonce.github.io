可等待执行: 在类内部维护一个队列数组queue，所有的函数执行都先加入数组，然后执行this.next()，通过next方法执行并销毁第一个函数，这样就可以传入可暂停的方法。

链式调用： 每个方法都返回类示例本身 return this

class PlayBoy {
  constructor(name) {
    this.name = name;
    this.queue = [];
    setTimeout(() => {
      console.log("执行");
      this.next();
    }, 0);
    return this;
  }

  next() {
    const fn = this.queue.shift();
    try {
      fn?.();
    } catch (error) {
      console.log({ fn });
    }
  }

  sayHi() {
    const fn = () => {
      console.log(this.name);
      this.next();
    };
    this.queue.push(fn);
    return this;
  }

  sleep(time) {
    const fn = () => {
      console.log(`等待了${time}秒`);
      setTimeout(() => {
        this.next();
      }, time);
    };
    this.queue.push(fn);
    return this;
  }

  play(name) {
    const fn = () => {
      console.log("i am playing", name);
      this.next();
    };
    this.queue.push(fn);

    return this;
  }
}

const pb = new PlayBoy("tom");
pb.sayHi().sleep(2000).play("王者").sleep(2000).play("热血航线");
